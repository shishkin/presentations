Live Projections

Map + Reduce + TransformResults

Map =
  users => from user in users
           select new {user.AliasId};

TransformResults =
  (database, users) =>
    from user in users
    let alias= database.Load(user.AliasId)
    select new {Name = user.Name, Alias = alias.Name};
